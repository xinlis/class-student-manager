第一次使用 Supabase 时，在 SQL Editor 执行下面这段建表语句。这个试用版靠“同步码”区分班级，请不要把同步码公开给别人。 create table if not exists public.classroom_sync ( id text primary key, payload jsonb not null, updated_at timestamptz not null default now() ); alter table public.classroom_sync enable row level security; create policy "classroom_sync_trial_select" on public.classroom_sync for select to anon using (id = current_setting('request.headers', true)::json ->> 'x-sync-id'); create policy "classroom_sync_trial_insert" on public.classroom_sync for insert to anon with check (id = current_setting('request.headers', true)::json ->> 'x-sync-id'); create policy "classroom_sync_trial_update" on public.classroom_sync for update to anon using (id = current_setting('request.headers', true)::json ->> 'x-sync-id') with check (id = current_setting('request.headers', true)::json ->> 'x-sync-id');
